To get started run the following SQL commands:


CREATE DATABASE misc;

GRANT ALL ON misc.* TO 'jitesh'@'localhost' IDENTIFIED BY 'hello';

GRANT ALL ON misc.* TO 'jitesh'@'127.0.0.1' IDENTIFIED BY 'hello';


USE misc; (Or select misc in phpMyAdmin)


CREATE TABLE users (
   
user_id INTEGER NOT NULL
  
   AUTO_INCREMENT KEY,
   
name VARCHAR(128),

   email VARCHAR(128),

   password VARCHAR(128),
  
 INDEX(email)
) ENGINE=InnoDB CHARSET=utf8;
